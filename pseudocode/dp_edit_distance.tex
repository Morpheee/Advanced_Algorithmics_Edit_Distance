		\begin{algorithm}[H]
		%\DontPrintSemicolon
		\KwInput{S1 , S2}
		\tcc{
			Parameters :\\
			\tab s1,s2 : strings.\\
			Return :\\
			\tab ED, alignment
		}
		\Begin{
			n \la len(s1) ; m \la len(s2)\\
			\textbf{define}~dist\_mat~:~array~with~first~row~0~to~m~and~first~column~0~to~n\\
			\textbf{define}~alignment~:~empty~array\\
			\If{(s1 == s2)}	
			{
				\Return{\{"ed" : 0 , "alignment" : "match"~for~all~letter \}}\;
			}
			\Else 
			{
				\For {j \la 1 to m+1}
				{
					\For {i \la 1 to n+1}
					{
						\If {s1[i-1] == s2[j-1]} 
						{
							dist\_mat[j,i] = dist\_mat[j-1 , i-1]\tcp{"keep" this letter}
						}
					}
					\Else {
						dist\_mat[j,i] = 1+min$
						\begin{cases}
						\text{dist\_mat[j,i-1]} \tcp{"remove" letter}\\
						\text{dist\_mat[j-1,i]} \tcp{"add" letter}\\
						\text{dist\_mat[j-1,i-1]} \tcp{"substitute" letter}
						\end{cases}$
					}
				}
			}
			alignment \la retrieve~alignment~by~backtracking~dist\_mat\;
			\Return{\{"ed" : dist\_mat[-1,-1] , "alignment" : alignment \}}\;
		}
		\caption{Dynamic\_Programming\_Single\_Matrix}
	\end{algorithm}
